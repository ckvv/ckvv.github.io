nodejs stream to buffer

/**
 * 
 * @param {ReadableStream} stream 
 */
async function streamToBuffer(stream) {
  return new Promise((resolve, reject) => {
    const buffers = [];

    stream.on('data', (chunk) => {
      buffers.push(chunk);
    });
    stream.on('end', () => {
      resolve(Buffer.concat(buffers))
    })

    stream.on('error', (error) => {
      reject(error)
    })
  });
}
